if, for

    var v = 'global.v'
    console.log(v)  //global.v
    if(true){
        var v = 'if.v'
        console.log(v)  //if.v 覆盖全局变量
    }
    console.log(v)  //if.v

    var v = 'global.v'
    console.log(v)  //global.v
    if(true){           //没有块级作用域
        (function(){    //模拟块级作用域
            var v = 'if.v'
            console.log(v)  //if.v
        })()
    }
    console.log(v)  //global.v
