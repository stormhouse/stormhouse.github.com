代码

html:

<!--el-->
<input id="datebox-el"/>

js:

$('#select').datebox();

描述

日期选择框，添加区域选择功能，添加align属性

配置

//未加注释，为原生配置
$('date-el').datebox({
  align: right,                        //下拉面板对齐方式，right
  currentDate: new Date('11/11/2011'), //初始日期
  range: true,                         //是否可选时间段
  startIndex: '09/18/2013',            //最小时间
  onSetValue: function(date){          //
    console.log(date);
  }
});

方法

$('#select').datebox('disable');
$('#select').datebox('enable');
//默认调用onSelect方法
$('#select').datebox('setDate', new Date());
//不会调用onSelect回调
$('#select').datebox('setDateNoCall', new Date());//or 10/28/2013-11/11/2013
$('#select').datebox('getDate');
$('#select').datebox('getText');
$('#select').datebox('setText', 'text');
$('#select').datebox('destroy');
